# Running the project

## Start Frontend / cert-app (port 5173/5174)
pnpm --filter @workspace/cert-app run dev

# Note on API URL:
# To use the live production API (no local wrangler needed):
#   Set VITE_API_URL=https://api.cephlow.in in apps/cert-app/.env.local (or env)
# To use the local API worker:
#   Set VITE_API_URL=http://localhost:8787 in apps/cert-app/.env.local (or env)

## Start Local API Worker (port 8787)
pnpm --filter @workspace/api-worker run dev

## URLs (local dev)
- Frontend:  http://localhost:5173
- API Local: http://localhost:8787
- API Prod:  https://api.cephlow.in

## URLs (production)
- Frontend: https://cephlow.in
- API:      https://api.cephlow.in

## Deploy API Worker to production
npx wrangler deploy --config=./apps/api-worker/wrangler.toml

## Sync Supabase data to D1 (if needed)
npx tsx scripts/db-sync-to-d1.ts
# Optionally dump data and execute SQL:
# npx wrangler d1 execute cephlow-app-db --remote --file=./scripts/d1-data-dump.sql --config=./apps/api-worker/wrangler.toml

## Database commands
# Apply schema locally
npx wrangler d1 execute cephlow-app-db --local --file=./apps/api-worker/schema.sql --config=./apps/api-worker/wrangler.toml

# Apply schema to production
npx wrangler d1 execute cephlow-app-db --remote --file=./apps/api-worker/schema.sql --config=./apps/api-worker/wrangler.toml

## Secrets management
# List secrets
npx wrangler secret list --config=./apps/api-worker/wrangler.toml

# Set a secret
echo "VALUE" | npx wrangler secret put SECRET_NAME --config=./apps/api-worker/wrangler.toml

## NOTE: apps/api-server is the OLD Digital Ocean Node.js server (no longer used).
## The active backend is apps/api-worker (Cloudflare Workers + D1 + R2).
